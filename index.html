function executeSQL() {
const query = document.getElementById('sql-editor').value;
const resultDiv = document.getElementById('result-output');
const errorDiv = document.getElementById('error-output');

resultDiv.innerHTML = "";
errorDiv.innerHTML = "";
try {
const results = db.exec(query);
if (results.length === 0) {
resultDiv.innerHTML = "<p>Success: Command executed (No rows returned).</p>";
return;
}
results.forEach(res => {
let table = "<table><thead><tr>";
res.columns.forEach(col => table += \`<th>\${col}</th>\`);
table += "</tr></thead><tbody>";
res.values.forEach(row => {
table += "<tr>";
row.forEach(val => table += \`<td>\${val ?? 'NULL'}</td>\`);
table += "</tr>";
});
table += "</tbody></table>";
resultDiv.innerHTML += table;
});
} catch (err) {
errorDiv.innerHTML = \`<div class="error">\${err.message}</div>\`;
}
}
